 54:     /**
 55:      * Returns the names of the enumeration constants of the specified
 56:      * enumeration type.
 57:      *
 58:      * The elements of the return value array are sorted in the order they are
 59:      * declared.
 60:      *
 61:      * This method may be used to iterate over the names of the enumerated
 62:      * constants as follows:
 63:      *
 64:      * <code>
 65:      * foreach (EnumUtils::names('Vendor\Package\ConcreteEnum') as $name) {
 66:      *     echo $name, \PHP_EOL;
 67:      * }
 68:      * </code>
 69:      *
 70:      * @param string $enumType The class name of the enumeration type from which
 71:      *                         to return the names of the enumeration constants.
 72:      *
 73:      * @return array An array containing the names of the enumeration constants
 74:      *               of the specified enumeration type, in the order they are
 75:      *               declared.
 76:      * @throws \InvalidArgumentException If the specified class name does not
 77:      *                                   represent an enumeration type.
 78:      */
 79:     public static function names($enumType = null)
 80:     {
 81:         self::_throwInvalidArgumentExceptionIfIsNotEnumType($enumType);
 82: 
 83:         return $enumType::names();
 84:     }

 86:     /**
 87:      * Returns an array containing the enumeration constants of the specified
 88:      * enumeration type, in the order they are declared.
 89:      *
 90:      * The elements of the return value array are sorted by the values of the

 92:      * the order of their corresponding names is unspecified.
 93:      *
 94:      * This method may be used to iterate over the enumerated constants as
 95:      * follows:
 96:      *
 97:      * <code>
 98:      * foreach (EnumUtils::values('Vendor\Package\ConcreteEnum') as $constant) {
 99:      *     echo $constant . \PHP_EOL;
100:      * }
101:      * </code>
102:      *
103:      * @param string $enumType The class name of the enumeration type from which
104:      *                         to return the enumeration constants.
105:      *
106:      * @return array An array containing the enumeration constants of the
107:      *               specified enumeration type, in the order they are declared.
108:      * @throws \InvalidArgumentException If the specified class name does not
109:      *                                   represent an enumeration type.
110:      */
111:     public static function values($enumType)
112:     {
113:         self::_throwInvalidArgumentExceptionIfIsNotEnumType($enumType);
114: 
115:         return $enumType::values();
116:     }

118:     /**
119:      * Throws an {@link \InvalidArgumentException} if the specified class name
120:      * does not represent an enumeration type.
121:      *
122:      * @param string $className The class name of the class to check.
123:      *
124:      * @return void
125:      * @throws \InvalidArgumentException If the specified class name does not
126:      *                                   represent an enumeration type.
127:      */
128:     private static function _throwInvalidArgumentExceptionIfIsNotEnumType(
129:         $className
130:     ) {
131:         if (false === self::isEnumType($className)) {
132:             throw new \InvalidArgumentException(
133:                 'The class ' . $className
134:                 . ' does not represent an enumeration type'
135:             );
136:         }

139:     /**

141:      *
142:      * @param string $className The class name of the class to check.
143:      *
144:      * @return boolean `true` if the specified class name represents an
145:      *                 enumeration type; `false` otherwise.
146:      */
147:     public static function isEnumType($className)
148:     {
149:         return \is_subclass_of($className, __NAMESPACE__ . '\EnumAbstract');
150:     }

152:     /**
153:      * Returns the enumeration constant of the specified enumeration type with
