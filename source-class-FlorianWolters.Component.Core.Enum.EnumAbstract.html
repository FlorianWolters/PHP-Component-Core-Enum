 35: /**
 36:  * The {@link EnumAbstract} class is the abstract base class of all PHP language
 37:  * type-safe enumerations.
 38:  *
 39:  * One feature of the C programming language lacking in PHP is enumerations.
 40:  * Using an implementation based on integers is poor and open to abuse. This
 41:  * class implements the *{@link
 42:  * http://java.sun.com/developer/Books/shiftintojava/page1.html#replaceenums
 43:  * Typesafe Enum}* pattern. The pattern has been adapted and abstracted for PHP.
 44:  *
 45:  * **Simple enumerations**
 46:  *
 47:  * To use this class, it must be subclassed. For example:
 48:  *
 49:  * <code>
 50:  * final class ColorEnum
 51:  *     extends FlorianWolters\Component\Core\Enum\EnumAbstract
 52:  * {
 53:  *     /** @return ColorEnum {@*}
 54:  *     final public static function RED() { return self::getConstant(); }
 55:  *     /** @return ColorEnum {@*}
 56:  *     final public static function GREEN() { return self::getConstant(); }
 57:  *     /** @return ColorEnum {@*}
 58:  *     final public static function BLUE() { return self::getConstant(); }
 59:  * }
 60:  * </code>
 61:  *
 62:  * Each enumeration constant has a name and an ordinal. These can be accessed
 63:  * using the {@link getName}, respectively the {@link getOrdinal} method.
 64:  *
 65:  * The {@link valueOf} and {@link values} methods can be used to access the
 66:  * enumeration constants dynamically.
 67:  *
 68:  * **Subclassed enumerations**
 69:  *
 70:  * A hierarchy of enumeration classes can be built. In this case, the superclass
 71:  * is unaffected by the addition of subclasses (as per normal PHP). The query
 72:  * methods on the subclass will return all of the enumeration constants from the
 73:  * superclass and the subclass.
 74:  *
 75:  * <code>
 76:  * // NOTE: The class ColorEnum declared above is final, change that to get this
 77:  * // example to run.
 78:  * final class ExtraColorEnum extends ColorEnum
 79:  * {
 80:  *     /** @return ExtraColorEnum {@*}
 81:  *     final public static function YELLOW() { return self::getConstant(); }
 82:  * }
 83:  * </code>
 84:  *
 85:  * **Functional enumerations**
 86:  *
 87:  * The enumeration classes can have functionality:
 88:  *
 89:  * <code>
 90:  * abstract class ArithmeticOperationEnum
 91:  *     extends FlorianWolters\Component\Core\Enum\EnumAbstract
 92:  * {
 93:  *     /** @return ArithmeticOperationEnum {@*}
 94:  *     final public static function PLUS() {
 95:  *         return static::getConstant(
 96:  *             'PlusOperationEnum', __FUNCTION__
 97:  *         );
 98:  *     }
 99:  *     /** @return ArithmeticOperationEnum {@*}
100:  *     final public static function MINUS() {
101:  *         return static::getConstant(
102:  *             'MinusOperationEnum', __FUNCTION__
103:  *         );
104:  *     }
105:  *     /** @return integer|float {@*}
106:  *     abstract public function evaluate($first, $second);
107:  * }
108:  *
109:  * final class PlusOperationEnum extends ArithmeticOperationEnum
110:  * {
111:  *     /** @return integer|float {@*}
112:  *     public function evaluate($first, $second) {
113:  *         return $first + $second;
114:  *     }
115:  * }
116:  *
117:  * final class MinusOperationEnum extends ArithmeticOperationEnum
118:  * {
119:  *     /** @return integer|float {@*}
120:  *     public function evaluate($first, $second) {
121:  *         return $first - $second;
122:  *     }
123:  * }
124:  * </code>
125:  *
126:  * Polymorphism is used to calculate the result of the artihmetic operation.
127:  *
128:  * <code>
129:  * $operations = [
130:  *     ArithmeticOperationEnum::PLUS(), ArithmeticOperationEnum::MINUS()
131:  * ];
132:  *
133:  * foreach ($operations as $operation) {
134:  *     echo $operation->evaluate(1, 1), \PHP_EOL;
135:  * }
136:  * </code>
137:  *
138:  * @category   Component
139:  * @package    Core
140:  * @subpackage Enum
141:  * @author     Florian Wolters <wolters.fl@gmail.com>
142:  * @copyright  2011-2012 Florian Wolters
143:  * @license    http://gnu.org/licenses/lgpl.txt LGPL-3.0+
144:  * @version    Release: @package_version@
145:  * @link       http://github.com/FlorianWolters/PHP-Component-Core-Enum
146:  * @since      Class available since Release 0.1.0
147:  */

214:     /**
215:      * Returns an array containing the names of the enumeration constants of
216:      * this enumeration type, in the order they are declared.
217:      *
218:      * This method may be used to iterate over the names of the enumerated
219:      * constants as follows:
220:      *
221:      * <code>
222:      * foreach (ConcreteEnum::names() as $name) {
223:      *     echo $name, \PHP_EOL;
224:      * }
225:      * </code>
226:      *
227:      * @return array An array containing the names of the constants of this
228:      *               enumeration type, in the order they are declared.
229:      */
230:     public static function names()
231:     {
232:         $enumType = \get_called_class();
233:         $parentEnumType = \get_parent_class($enumType);
234:         $parentMethods = [];
235: 
236:         if ((false !== $parentEnumType) && (__CLASS__ !== $parentEnumType)) {
237:             $parentMethods = $parentEnumType::names();
238:         }
239: 
240:         $reflectedClass = new \ReflectionClass($enumType);
241: 
242:         // The argument $filter of the method getMethods uses a logical OR.
243:         // Therefore we have to return all methods for each filter and calculate
244:         // the intersect.
245: 
246:         $finalMethods = $reflectedClass->getMethods(
247:             \ReflectionMethod::IS_FINAL
248:         );
249:         $staticMethods = $reflectedClass->getMethods(
250:             \ReflectionMethod::IS_STATIC
251:         );
252:         $publicMethods = $reflectedClass->getMethods(
253:             \ReflectionMethod::IS_PUBLIC
254:         );
255:         $reflectedMethods = \array_intersect(
256:             $finalMethods, $staticMethods, $publicMethods
257:         );
258: 
259:         $currentMethods = [];
260: 
261:         /* @var $method \ReflectionMethod */
262:         foreach ($reflectedMethods as $method) {
263:             if ($method->class === $enumType) {
264:                 $currentMethods[] = $method->name;
265:             }
266:         }
267: 
268:         $mergedMethods = \array_merge($parentMethods, $currentMethods);
269:         $result = \array_unique($mergedMethods);
270: 
271:         return $result;
272:     }

335:     /**
336:      * Creates and returns an enumeration constant for the specified enumeration
337:      * type with the specified name.
338:      *
339:      * This method implements the *Lazy Load* implementation pattern.
340:      *
341:      * Use this method as follows in a subclass of this class:
342:      *
343:      * <code>
344:      * final class ConcreteEnum
345:      *     extends FlorianWolters\Component\Core\Enum\EnumAbstract
346:      * {
347:      *     /** @return ConcreteEnum {@*}
348:      *     final public static function CONSTANT()
349:      *     {
350:      *         return self::getConstant();
351:      *     }
352:      * }
353:      * </code>
354:      *
355:      * @param string $enumType The name of the enumeration type.
356:      * @param string $name     The name of the enumeration constant, which is
357:      *                         the identifier used to declare it.
358:      *
359:      * @return EnumAbstract The enumeration constant of the specified
360:      *                      enumeration type with the specified name.
361:      * @throws \InvalidArgumentException If the specified enumeration type has
362:      *                                   no enumeration constant with the
363:      *                                   specified name, or the specified class
364:      *                                   name does not represent an enumeration
365:      *                                   type.
366:      */
367:     final protected static function getConstant($enumType = null, $name = null)
368:     {
369:         if ((null === $enumType) || (null === $name)) {
370:             // TODO Is there a faster solution to retrieve the calling class and
371:             // calling method?
372:             // TODO This does not allow late-static binding.
373:             $backtrace = \debug_backtrace(\DEBUG_BACKTRACE_IGNORE_ARGS);
374:             $enumType = $backtrace[1]['class'];
375:             $name = $backtrace[1]['function'];
376:         } else {
377:             // TODO Check the plausability of the specified parameters? Or leave
378:             // it as it is as a "performance mode"?
379:             $enumType = (string) $enumType;
380:             $name = (string) $name;
381:             self::_exists($enumType, $name);
382:         }
383: 
384:         if (false === isset(self::$_constants[$enumType][$name])) {
385:             $ordinal = self::_retrieveOrdinalFor($name);
386:             $newInstance = new $enumType($name, $ordinal);
387: 
388:             self::$_constants[$enumType][$name] = $newInstance;
389:         }
390: 
391:         return self::$_constants[$enumType][$name];
392:     }

395:      * Checks whether the enumeration constant with the specified name exists in
396:      * the specified enumeration type.
397:      *
398:      * @param string $enumType The name of the enumeration type.
399:      * @param string $name     The name of the enumeration constant, which is
400:      *                         the identifier used to declare it.

403:      * @throws \InvalidArgumentException If the specified enumeration type has

407:     private static function _exists($enumType, $name)
408:     {
409:         // TODO This is not 100% exact as the client can swap the name of two
410:         // enumeration constants without raising an exception. Should be fixed
411:         // in 1.0.0. Possible fix would be using the backtrace and comparing the
412:         // caller with the specified name.
413:         if (false === \method_exists($enumType, $name)) {
414:             throw new \InvalidArgumentException(
415:                 'No enumeration constant ' . $enumType . '::' .  $name . '()'
416:             );
417:         }
418:     }

421:      * Returns the ordinal of the enumeration constant with the specified name
422:      * in this enumeration type.
423:      *
424:      * @param string $name The name of the enumeration constant, which is the
425:      *                     identifier used to declare it.

428:      *                 the enumeration declaration, where the initial

432:     {
433:         $names = self::names();
434:         $result = null;

440:             }

661:     /**
662:      * Determines whether the specified enumeration constants are equal.
663:      *
664:      * Two enumeration constants are considered equal if they have the same
665:      * class names, the same names and the same ordinals.
666:      *
667:      * **Most programmers should use the *identical comparison operator* (`===`)
668:      * in preference to this method, as the comparison operator is faster.**
669:      *
670:      * <code>
671:      * ConcreteEnum::FIRST_CONSTANT() === ConcreteEnum::SECOND_CONSTANT()
672:      * </code>
673:      *
674:      * @param EnumAbstract $first  The first enumeration constant to compare.
675:      * @param EnumAbstract $second The second enumeration constant to compare.
676:      *
677:      * @return boolean `true` if the enumeration constants are equal; `false`
678:      *                 otherwise.
679:      * @deprecated Use the *identical comparison operator* (`===`) instead.
680:      */
681:     final static public function isEqual(self $first, self $second)
682:     {
683:         return ($first->_className === $second->_className)
684:             && ($first->_name === $second->_name)
